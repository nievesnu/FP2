int nuevo, pos;
// Desde el segundo elemento hasta el último...
for (int i = 1; i < N; i++) {
    nuevo = lista[i];
    pos = 0;
    while ((pos < i) && (lista[pos] > nuevo)) {
        pos++;
    }
    // pos: índice del primer mayor; i si no lo hay
    for (int j = i; j > pos; j--) {
        lista[j] = lista[j - 1];
    }
    lista[pos] = nuevo;
}
